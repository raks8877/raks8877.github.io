{% highlight c++ %} {% endhighlight %} {% highlight c++ %} ListNode *add_number(ListNode *l1, ListNode *l2, int carry) { if(l1 == NULL && l2 == NULL && carry == 0) return NULL; int sum = (l1 == NULL?0:l1->val) + (l2 == NULL?0:l2->val) + carry; carry = sum / 10; sum %= 10; ListNode *head = new ListNode(sum); head->next = add_number(l1 == NULL?l1:l1->next, l2 == NULL?l2:l2->next, carry); return head; } {% endhighlight %}
